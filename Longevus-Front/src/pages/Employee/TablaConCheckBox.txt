import Table from '../../components/TableBasic';
import type {columnDefinition} from '../../components/TableBasic';
import Footer from '../../components/Footer';
import Header from '../../components/Header';
import { useState, useMemo} from 'react';
import { Link } from 'react-router-dom';

interface IPerson{
    id: number,
    name: string,
    identification: string,
    email: string,
}
const userData: IPerson[] = [
  { id: 1, name: 'Alice Smith',identification: '3423242' ,email: 'alice@example.com'},
  { id: 4, name: 'Karla ', identification: 't65fg5gs',email: 'bob@example.com'  },
  { id: 7, name: 'Harry', identification: 't65fg5gs',email: 'bob@example.com'  },
  { id: 10, name: 'Krystal', identification: 't65fg5gs',email: 'bob@example.com'  },
  // ... más usuarios
];
const ShowEmployee = () =>{

    const [selectedRowIds, setSelectedRowIds] = useState<Set<number>>(new Set());
    // Función para (de)seleccionar una sola fila
  const handleToggleRow = (rowId: number) => {
    setSelectedRowIds(prevSelectedIds => {
      const newSelectedIds = new Set(prevSelectedIds);
      if (newSelectedIds.has(rowId)) {
        newSelectedIds.delete(rowId);
      } else {
        newSelectedIds.add(rowId);
      }
      return newSelectedIds;
    });
  };

  // Función para (de)seleccionar todas las filas
  const handleSelectAll = (isSelected: boolean) => {
    if (isSelected) {
      // Si se selecciona todo, añade todos los IDs del initialData al set
      const allIds = new Set(userData.map(person => person.id));
      setSelectedRowIds(allIds);
    } else {
      // Si se deselecciona todo, vacía el set
      setSelectedRowIds(new Set());
    }
  };

  const checkboxColumn: columnDefinition<IPerson> = useMemo(() => ({
    header: 'Seleccionar', // Texto por defecto (no visible si se usa Header)
    // El Header renderiza el checkbox "Seleccionar Todo"
    Header: ({ isSelectedAll, onSelectAll }) => ( // Recibe las props que le pasamos desde GenericTable
      <input
        type="checkbox"
        checked={isSelectedAll}
        onChange={(e) => onSelectAll(e.target.checked)} // Llama al manejador del padre
      />
    ),
    accessor: (person) => person.id, // El accessor podría ser el ID de la fila
    Cell: (person) => ( // Renderiza el checkbox individual en cada fila
      <input
        type="checkbox"
        checked={selectedRowIds.has(person.id)} // Está marcado si su ID está en el set
        onChange={() => handleToggleRow(person.id)} // Llama al manejador del padre con el ID de esta fila
      />
    ),
  }), [selectedRowIds]); // Regenerar esta definición si cambia selectedRowIds (necesario para 'checked')


   const personColumns: columnDefinition<IPerson>[] = useMemo(()=>[
    checkboxColumn,
    {header: '#', accessor: 'id', Cell:(person, index)=>{return(index+1)}},
    {header: 'Nombre', accessor: 'name'},
    {header: 'Identificacion', accessor: 'identification'},
    {header: 'Correo', accessor: 'email'},
    {header: 'Acciones', accessor: (person) => person,   
        Cell: (person) =>(
            <>
            <a className='btn btn-info me-2' onClick={()=>console.log("Mostrar"+person.id)}>
                <i className='bi bi-eye'/>
            </a>
            <a className='btn btn-warning me-2' onClick={()=>console.log("Editar"+person.name)}>
                <i className='bi bi-pencil-square'/>
            </a>
            
            <a className='btn btn-danger me-2' onClick={()=>console.log("Eliminar"+person.id)}>
                <i className="bi bi-trash"/>
            </a>
            
            </>
        )
        
    }
   ], [checkboxColumn]);

   return(
    <>
    <Header/>
    <div className='container'>
            <div className='row'>
                <div className='card mt-5 mb-5'>
                    <div className='card-title d-flex justify-content-between align-items-center mt-3'>
                        <h4>Lista de empleados</h4>
                        <Link className='btn btn-success' to='/agregar'>Agregar</Link>
                    </div>
                    <div className='card-body'>
                        <label>Buscar</label>
                        <input type="text" placeholder="Buscar..." id="userSearch"/>
                        <button className="btn btn-secondary" id="btnSearch"><i className='bi bi-search'/></button>
                        <Table<IPerson> data={userData} columns={personColumns} selectedRows={selectedRowIds} onToggleRow={handleToggleRow} onSelectAll={handleSelectAll}/>
                         <p>Filas seleccionadas: {selectedRowIds.size}</p>
               
                        {selectedRowIds.size > 0 && (
                             <button className="btn btn-light" onClick={() => console.log("Procesar IDs:", Array.from(selectedRowIds))}>
                                    Procesar Seleccionados ({selectedRowIds.size})
                        </button>)}
                    </div>
                </div>
                
            </div>
            
        </div>
        <Footer/>
    </>
        
        
        
   )


   
}
export default ShowEmployee;
